// traverse the json of the tags
function traverse(tags)
{
	for(item in tags)
	{
		var val = tags[item];
		// if its a table run the table rule on it
		if(item.toLowerCase() == "table")
		{
			tableformatted(val);
		}
		else if(val.constructor == Array)
		{
			for(i=0;i<val.length;i++)
			{
				traverse(val[i]);
			}
		}
		else{
			traverse(val);
		}
	}
}

function contains(list, str)
{

	}
	return false;
